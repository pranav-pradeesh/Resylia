## Security Architecture
> This section is non-negotiable. Mental health data is among the most sensitive data a company can hold. A breach does not just lose you customers — it ends the company and causes real harm to real people. Every pattern below is enforced from day one, not added later.

---

### Threat Model — What We Are Actually Defending Against

```
1. Prompt Injection         — user submits malicious text through check-in free-text
                              to manipulate the AI coach or extract other users' data
2. IDOR                     — user manipulates IDs in API calls to access other orgs' data
3. Broken Access Control    — employee accesses manager routes, manager accesses HR routes
4. Webhook Replay Attacks   — attacker replays Stripe/Slack webhooks to trigger actions
5. API Abuse / DoS          — flooding Claude API endpoints to exhaust budget or crash service
6. Data Exfiltration via AI — crafted prompts that trick AI into revealing aggregated patterns
7. Session Hijacking        — stolen JWT used to impersonate employees
8. SQL Injection            — malicious input reaching database queries
9. XSS via AI Output        — AI returns HTML/JS in response that gets rendered unsanitized
10. Sensitive Data Leakage  — mental health scores leaked in logs, errors, or API responses
```

### Layer 1 — AI-Specific Security (Highest Priority)

This is where most AI products get compromised. The free-text check-in field is a direct injection surface into the Claude API.

#### 1.1 Prompt Injection Defense

```typescript
// packages/ai/src/sanitize.ts

const INJECTION_PATTERNS = [
  /ignore (previous|above|all) instructions/i,
  /you are now/i,
  /system prompt/i,
  /\[INST\]/i,
  /###\s*(system|instruction)/i,
  /<\|im_start\|>/i,
  /forget (everything|all|your instructions)/i,
  /pretend (you are|to be)/i,
  /roleplay as/i,
  /disregard/i,
  /jailbreak/i,
  /DAN mode/i,
];

export function sanitizeUserInput(input: string): {
  safe: boolean;
  sanitized: string;
  flagged: boolean;
} {
  // 1. Length cap — never send more than 500 chars to AI
  const truncated = input.slice(0, 500);

  // 2. Strip HTML entirely — never allow markup in AI input
  const stripped = truncated.replace(/<[^>]*>/g, '');

  // 3. Check for injection patterns
  const flagged = INJECTION_PATTERNS.some(p => p.test(stripped));

  // 4. If flagged: log the attempt, do NOT send to Claude, return safe fallback
  if (flagged) {
    logSecurityEvent('prompt_injection_attempt', { input: truncated });
    return { safe: false, sanitized: '', flagged: true };
  }

  return { safe: true, sanitized: stripped, flagged: false };
}
```

#### 1.3 AI Output Sanitization

Never render AI output directly as HTML. Always sanitize before display.

```typescript
// packages/ai/src/output.ts
import DOMPurify from 'isomorphic-dompurify';

export function sanitizeAIOutput(raw: string): string {
  // 1. Parse as JSON first — if it fails, the response is malformed
  let parsed: unknown;
  try {
    parsed = JSON.parse(raw);
  } catch {
    logSecurityEvent('ai_malformed_output', { raw: raw.slice(0, 200) });
    throw new Error('AI returned invalid response format');
  }

  // 2. Validate schema — only expected fields allowed
  if (!isValidCoachResponse(parsed)) {
    throw new Error('AI response failed schema validation');
  }

  // 3. Sanitize string fields — strip any HTML that snuck through
  const result = parsed as CoachResponse;
  return {
    suggestion: DOMPurify.sanitize(result.suggestion, { ALLOWED_TAGS: [] }),
    category: result.category.replace(/[^a-z_]/gi, ''),  // alphanumeric only
    duration_minutes: Math.min(Math.max(Number(result.duration_minutes), 1), 60),
  };
}
```

#### 1.4 Per-Org Claude API Budget Cap

Prevents a single org or malicious user from exhausting your entire Claude API budget.

```typescript
// apps/web/app/api/_middleware/ai-budget.ts
import { redis } from '@/lib/redis';

const DAILY_LIMITS = {
  starter: 100,    // max AI calls per day per org
  growth: 500,
  enterprise: 2000,
};

export async function checkAIBudget(orgId: string, plan: string): Promise<void> {
  const key = `ai_budget:${orgId}:${new Date().toISOString().split('T')[0]}`;
  const count = await redis.incr(key);

  if (count === 1) {
    await redis.expire(key, 86400); // expires in 24h
  }

  const limit = DAILY_LIMITS[plan as keyof typeof DAILY_LIMITS] ?? 100;

  if (count > limit) {
    throw new TooManyRequestsError(`Daily AI usage limit reached for your plan`);
  }
}
```

#### 3.2 Server-Side Authorization Middleware

```typescript
// apps/web/app/api/_middleware/auth.ts

import { createServerClient } from '@supabase/ssr';
import { NextRequest, NextResponse } from 'next/server';

export async function withAuth(
  req: NextRequest,
  handler: (userId: string, orgId: string, role: string) => Promise<NextResponse>,
  requiredPermission: keyof typeof PERMISSIONS
): Promise<NextResponse> {
  const supabase = createServerClient(/* ... */);

  // 1. Validate JWT
  const { data: { user }, error } = await supabase.auth.getUser();
  if (error || !user) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  // 2. Fetch user's org and role from DB (never trust JWT claims for role)
  const { data: profile } = await supabase
    .from('users')
    .select('org_id, role, is_active')
    .eq('id', user.id)
    .single();

  if (!profile || !profile.is_active) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
  }

  // 3. Check permission
  const allowed = PERMISSIONS[requiredPermission].includes(profile.role);
  if (!allowed) {
    logSecurityEvent('unauthorized_access_attempt', {
      userId: user.id,
      orgId: profile.org_id,
      permission: requiredPermission,
      path: req.nextUrl.pathname,
    });
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 });
  }

  // 4. Pass verified identifiers to handler — never use user-supplied org_id
  return handler(user.id, profile.org_id, profile.role);
```

### Layer 4 — Supabase Row Level Security (RLS)

RLS is your last line of defense if application-layer auth fails. Every table must have RLS enabled.

```sql
-- supabase/migrations/20260422_rls.sql

-- Enable RLS on all tables
ALTER TABLE organizations ENABLE ROW LEVEL SECURITY;
ALTER TABLE users ENABLE ROW LEVEL SECURITY;
ALTER TABLE checkins ENABLE ROW LEVEL SECURITY;
ALTER TABLE risk_events ENABLE ROW LEVEL SECURITY;
ALTER TABLE alerts ENABLE ROW LEVEL SECURITY;
ALTER TABLE subscriptions ENABLE ROW LEVEL SECURITY;

-- Users: can only see their own org
CREATE POLICY "users_own_org" ON users
  FOR ALL USING (
    org_id = (SELECT org_id FROM users WHERE id = auth.uid())
  );

-- Check-ins: employees see only their own
-- Managers/HR see only aggregated (enforced at app layer + this blocks raw access)
CREATE POLICY "checkins_own_only" ON checkins
  FOR SELECT USING (user_id = auth.uid());

CREATE POLICY "checkins_insert_own" ON checkins
  FOR INSERT WITH CHECK (
    user_id = auth.uid() AND
    org_id = (SELECT org_id FROM users WHERE id = auth.uid())
  );

-- Risk events: only viewable by managers/HR within same org
CREATE POLICY "risk_events_org_managers" ON risk_events
  FOR SELECT USING (
    org_id = (SELECT org_id FROM users WHERE id = auth.uid()) AND
    (SELECT role FROM users WHERE id = auth.uid()) IN ('manager', 'hr', 'admin')
  );

-- Subscriptions: only org admins
CREATE POLICY "subscriptions_admin_only" ON subscriptions
  FOR ALL USING (
    org_id = (SELECT org_id FROM users WHERE id = auth.uid()) AND
    (SELECT role FROM users WHERE id = auth.uid()) = 'admin'
  );

-- Service role bypasses RLS (used only in server-side code, never exposed to client)
-- SUPABASE_SERVICE_ROLE_KEY must NEVER be sent to the browser
```

### Layer 6 — Webhook Security

Stripe and Slack webhooks must be verified. An unverified webhook is a free remote code execution vector.

```typescript
// apps/web/app/api/webhooks/stripe/route.ts

import Stripe from 'stripe';

export async function POST(req: NextRequest) {
  const body = await req.text();  // raw body — must be text, not parsed JSON
  const sig = req.headers.get('stripe-signature');

  if (!sig) {
    return NextResponse.json({ error: 'Missing signature' }, { status: 400 });
  }

  let event: Stripe.Event;
  try {
    // Stripe verifies both the signature AND the timestamp to prevent replay attacks
    event = stripe.webhooks.constructEvent(body, sig, process.env.STRIPE_WEBHOOK_SECRET!);
  } catch {
    logSecurityEvent('invalid_stripe_webhook', { sig });
    return NextResponse.json({ error: 'Invalid signature' }, { status: 400 });
  }

  // Process only known event types — ignore everything else
  const HANDLED_EVENTS = [
    'customer.subscription.created',
    'customer.subscription.updated',
    'customer.subscription.deleted',
    'invoice.payment_failed',
  ];

  if (!HANDLED_EVENTS.includes(event.type)) {
    return NextResponse.json({ received: true });
  }

  // Handle event...
}
```

### Layer 8 — Sensitive Data Handling

Mental health check-in data requires special treatment beyond standard PII.

```typescript
// packages/db/src/sensitive.ts

// Fields that must NEVER appear in logs, error messages, or API responses
// to unauthorized roles
export const SENSITIVE_FIELDS = [
  'energy',
  'stress',
  'workload',
  'sentiment_score',
  'burnout_risk_score',
  'free_text',
  'contributing_factors',
  'predicted_burnout_date',
] as const;

// Redact sensitive fields from error logs
export function redactForLog(obj: Record<string, unknown>): Record<string, unknown> {
  return Object.fromEntries(
    Object.entries(obj).map(([k, v]) => [
      k,
      SENSITIVE_FIELDS.includes(k as any) ? '[REDACTED]' : v
    ])
  );
}

// API responses to managers strip individual-level sensitive data
export function redactForManager(checkins: Checkin[]): AggregatedTeamData {
  // Never return raw checkins to managers — only aggregated stats
  // Minimum 5 members enforced here AND at RLS level
  if (checkins.length < 5) {
    return { insufficient_data: true };
  }
  return {
    avg_stress: average(checkins.map(c => c.stress)),
    avg_energy: average(checkins.map(c => c.energy)),
    high_risk_count: checkins.filter(c => c.burnout_risk_score > 0.7).length,
    // Individual scores, names, and free_text are never included
  };
}
```

### Layer 9 — Audit Logging

Every security-relevant event must be logged. Required for SOC 2 and enterprise contracts.

```typescript
// packages/shared/src/audit.ts

type AuditEvent =
  | 'prompt_injection_attempt'
  | 'unauthorized_access_attempt'
  | 'invalid_stripe_webhook'
  | 'rate_limit_exceeded'
  | 'ai_malformed_output'
  | 'login_failed'
  | 'role_escalation_attempt'
  | 'data_export'
  | 'user_deactivated'
  | 'billing_change';

export async function logSecurityEvent(
  event: AuditEvent,
  context: Record<string, unknown>
): Promise<void> {
  // 1. Never log sensitive field values in audit logs
  const sanitized = redactForLog(context);

  // 2. Write to append-only audit_log table in Supabase (service role only)
  await adminDb.from('audit_log').insert({
    event,
    context: sanitized,
    ip: getRequestIP(),
    created_at: new Date().toISOString(),
  });

  // 3. For critical events, also send to Sentry with high severity
  if (['prompt_injection_attempt', 'unauthorized_access_attempt', 'role_escalation_attempt'].includes(event)) {
    Sentry.captureMessage(`Security event: ${event}`, {
      level: 'warning',
      extra: sanitized,
    });
  }
}
```
